  <script>
    const apiKey = "ВАШ_API_КЛЮЧ"; // Вставь сюда свой YouTube API ключ

    const videoBlocks = {
      blockAlykul: ["AOuj_1hr264", "xUqTq9UPnnk", "G2seyNojp8M", "GJKPy7FGR3U"],
      blockAytmatov: ["gaK1s-agCNE", "yL8EMBMxjCA", "RJNZJZOEH5A", "Z4T_E_yuWdY"]
    };

    async function fetchVideoTitle(id) {
      const response = await fetch(`https://www.googleapis.com/youtube/v3/videos?part=snippet&id=${id}&key=${apiKey}`);
      const data = await response.json();
      if (data.items && data.items.length > 0) {
        return data.items[0].snippet.title;
      }
      return "Нет названия";
    }

    async function loadVideos() {
      for (const [blockId, ids] of Object.entries(videoBlocks)) {
        const container = document.getElementById(blockId);
        for (const id of ids) {
          const title = await fetchVideoTitle(id);
          const card = document.createElement("div");
          card.className = "video-card";
          card.onclick = () => playVideo(id);
          card.innerHTML = `
            <img src="https://img.youtube.com/vi/${id}/hqdefault.jpg" alt="${title}">
            <div>${title}</div>
          `;
          container.appendChild(card);
        }
      }
    }

    function playVideo(id) {
      document.getElementById('videoPlayer').src = 'https://www.youtube.com/embed/' + id + '?autoplay=1';
      window.scrollTo({ top: document.getElementById('videoPlayer').offsetTop - 20, behavior: 'smooth' });
    }

    loadVideos();
  </script>
